import json as _json, os as _os

_LOOKUP_TABLE: list[dict] | None = None

def _load_ts_lookup() -> list[dict]:
    global _LOOKUP_TABLE
    if _LOOKUP_TABLE is None:
        path = _os.path.join(_os.path.dirname(__file__), "recovery_to_strain.json")
        try:
            with open(path) as fh:
                _LOOKUP_TABLE = _json.load(fh)
        except OSError:
            logging.warning("recovery_to_strain.json not found — training_state will be None")
            _LOOKUP_TABLE = []
    return _LOOKUP_TABLE

def training_state_from_lookup(recovery_score: float | None, strain: float | None) -> str | None:
    """ALG-11: Training State server-side via recovery_to_strain.json lookup.
    recovery_score deve ser 0–100 (não 0–1 fraction).
    Retorna 'RESTORATIVE' | 'OPTIMAL' | 'OVERREACHING' | None."""
    if recovery_score is None or strain is None:
        return None
    table = _load_ts_lookup()
    if not table:
        return None
    idx = int(round(min(100.0, max(0.0, recovery_score))))
    row = next((r for r in table if r.get("recovery") == idx), table[-1])
    lower = row.get("lower_rec_strain", 0.0)
    upper = row.get("upper_rec_strain", 21.0)
    if strain < lower:
        return "RESTORATIVE"
    elif strain > upper:
        return "OVERREACHING"
    else:
        return "OPTIMAL"
```

def sleep_needed(
    prior_sleep_min: list[float],
    strain_yesterday: float | None,
    sleep_yesterday: float | None,
) -> float | None:
    """ALG-12: Sleep Needed = baseline rolling 7d + strain debt + sleep debt.
    prior_sleep_min: lista de total_sleep_min dos últimos 7d (excluindo hoje); pode ter None removidos.
    Retorna minutos [300, 660] ou None se baseline insuficiente (< 3 noites válidas)."""
    valid = [v for v in prior_sleep_min if v is not None and v > 0]
    if len(valid) < 3:
        return None
    baseline = statistics.mean(valid)

    # Strain debt: +3 min por unidade de strain acima de 14, max 60 min (per ALG-12 spec)
    strain_debt = 0.0
    if strain_yesterday is not None and strain_yesterday > 14.0:
        strain_debt = min(60.0, (strain_yesterday - 14.0) * 3.0)

    # Sleep debt: défice acumulado da noite anterior, máx 120 min, contribui 50%
    sleep_debt = 0.0
    if sleep_yesterday is not None:
        raw_debt = max(0.0, baseline - sleep_yesterday)
        sleep_debt = min(raw_debt, 120.0) * 0.5

    needed = baseline + strain_debt + sleep_debt
    return round(max(300.0, min(660.0, needed)), 1)
```
